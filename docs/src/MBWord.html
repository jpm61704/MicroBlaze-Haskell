{-# LANGUAGE UnicodeSyntax #-}
module MBWord where

import           Data.Word

newtype MBWord = MBWord Word32 deriving (Show)

newtype CarryBit = CarryBit Bool


add ∷ MBWord → MBWord → CarryBit → MBWord
add (MBWord x) (MBWord y) (CarryBit carry) = MBWord $ x + y + c
  where c = if carry then 1 else 0


w1 ∷ MBWord
w1 = MBWord $ fromIntegral 1


w2 ∷ MBWord
w2 = MBWord $ fromIntegral 2
